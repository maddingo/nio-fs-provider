27  @RunWith(Parameterized.class)
28  public class DirectoryStreamIteratorTest extends AbstractSmbTest {
29  
30      private String[] childrenUrls;
31      private String parentUrl;
32      private String[] childrenUris;
33      private String parentUri;
34  
35      @Parameterized.Parameters(name ="{index} {0}")
36      public static Iterable<Object[]> data() {
37          List<Object[]> data = new ArrayList<>();
38  
39          data.add(new Object[] {
40                  "smb://127.0.0.1/public/My Documents/",
41                  new String[] {"smb://127.0.0.1/public/My Documents/Folder One/", "smb://127.0.0.1/public/My Documents/Folder Two/"}
42          });
43          return data;
44      }
45  
46      public DirectoryStreamIteratorTest(String parentUrl, String[] childrenUrls) throws MalformedURLException, URISyntaxException {
47          this.parentUrl = parentUrl;
48          this.childrenUrls = childrenUrls;
49          URL parent = new URL(parentUrl);
50          this.parentUri = new URI(parent.getProtocol(), parent.getAuthority(), parent.getPath(), null).toString();
51          this.childrenUris = new String[childrenUrls.length];
52          for (int i = 0; i < childrenUrls.length; i++) {
53              URL child = new URL(childrenUrls[i]);
54              this.childrenUris[i] = new URI(child.getProtocol(), child.getAuthority(), child.getPath(), null).toString();
55          }
56      }
57  
58      @Rule
59      public ExpectedException exception = ExpectedException.none();
60  
61      @Test
62      public void directoryStreamIterator() throws Exception {
63  
64          // ignore the test if the expected test file system does not exist
65          for (String child : childrenUrls) {
66              SmbFile smbf = new SmbFile(child);
67              assumeThat("Test file " + smbf.toString() + " exists", smbf.canRead(), is(true));
68          }
69  
70          Path remotePath = Paths.get(new URI(parentUri));
71          List<String> fileNames = new ArrayList<>();
72          try (DirectoryStream<Path> directoryStream = Files.newDirectoryStream(remotePath, new DirectoryStream.Filter<Path>() {
73              @Override
74              public boolean accept(Path entry) throws IOException {
75                  return !Files.isSymbolicLink(entry);
76              }
77          })) {
78              for (Path path : directoryStream) {
79                  fileNames.add(path.toString());
80              }
81          }
82  
83          assertThat(fileNames, hasItems(childrenUris));
84      }
85  }
