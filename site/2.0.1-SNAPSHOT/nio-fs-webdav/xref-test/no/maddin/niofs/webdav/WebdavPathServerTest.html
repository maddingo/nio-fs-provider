35      /**
36       * Start an embedded Webdav Server for testing.
37       */
38      @Before
39      public void startWebdavServer() throws Exception {
40          rootFolder = Files.createTempDirectory(Paths.get(".", "target"), "webdav-test").toFile();
41          server = new MiltonWebDAVFileServer(rootFolder);
42          try (ServerSocket ssock = new ServerSocket(0, 1, new InetSocketAddress(0).getAddress())) {
43              webdavPort = ssock.getLocalPort();
44              server.setPort(webdavPort); // optional, defaults to 8080
45              server.getUserCredentials().put(user, password); // optional, defaults to no authentication
46              ssock.close(); // close before the server can re-bind to the random port.
47              server.start();
48          }
49      }
50  
51      /**
52       * Stop the embedded Webdav server.
53       */
54      @After
55      public void stopWebDavServer() throws Exception {
56          server.stop();
57      }

59      /**
60       * Tests {@link Files#createDirectories(Path, FileAttribute[])} with WebDavPath
61       */
62      @Test
63      public void getCreateChildPath() throws Exception {
64          URI uri = new URI("webdav", user + ':' + password,"localhost", webdavPort, "/a/b", null, null);
65          Path path = Paths.get(uri);
66          Path newPath = Files.createDirectories(path);
67          assertThat(newPath, is(notNullValue()));
68          File fileA = new File(rootFolder.getAbsolutePath(), "a");
69          assertThat(fileA.exists(), is(true));
70          assertThat(fileA.isDirectory(), is(true));
71          File fileB = new File(fileA.getAbsolutePath(), "b");
72          assertThat(fileB.exists(), is(true));
73          assertThat(fileB.isDirectory(), is(true));
74      }

76      /**
77       * Tests {@link Files#copy(Path, Path, CopyOption...)} with WebdavPath as target.
78       */
79      @Test
80      public void copyFiles() throws Exception {
81          File src = File.createTempFile("webdavtest", ".txt", new File("target"));
82          try (FileWriter fw = new FileWriter(src)) {
83              fw.append("test test");
84          }
85  
86          URI uriTo = new URI("webdav", user + ':' + password,"localhost", webdavPort, "/webdav/test2/file.txt", null, null);
87  
88          Path pathTo = Paths.get(uriTo);
89          Files.copy(src.toPath(), pathTo, StandardCopyOption.REPLACE_EXISTING);
90  
91          File fileWebdav = new File(rootFolder.getAbsolutePath(), "webdav");
92          assertThat(fileWebdav.exists(), is(true));
93          assertThat(fileWebdav.isDirectory(), is(true));
94          File fileTest2 = new File(fileWebdav.getAbsolutePath(), "test2");
95          assertThat(fileTest2.exists(), is(true));
96          assertThat(fileTest2.isDirectory(), is(true));
97          File fileTxt = new File(fileTest2.getAbsolutePath(), "file.txt");
98          assertThat(fileTxt.exists(), is(true));
99          assertThat(fileTxt.isFile(), is(true));
100 
101         StringBuilder content = new StringBuilder();
102         try (BufferedReader br = new BufferedReader(new FileReader(fileTxt))) {
103             content.append(br.readLine());
104         }
105         assertThat(content.toString(), is(equalTo("test test")));
106     }

118     @Test
119     public void deleteFile() throws Exception {
120         File targetFile = new File(rootFolder.getAbsolutePath(), "webdav/test2/file.txt");
121         assertThat(targetFile.getParentFile().mkdirs(), is(true));
122         try (FileWriter fw = new FileWriter(targetFile)) {
123             fw.append("test test, delete file");
124         }
125         URI uri = new URI("webdav", user + ':' + password,"localhost", webdavPort, "/webdav/test2/file.txt", null, null);
126         Path path = Paths.get(uri);
127         assumeThat("File should exist prior to deleting it", Files.exists(path), is(true));
128 
129         // delete the file
130         Files.delete(path);
131 
132         assumeThat("File " + targetFile.getAbsolutePath() + " should have perished", targetFile.exists(), is(false));
133     }
