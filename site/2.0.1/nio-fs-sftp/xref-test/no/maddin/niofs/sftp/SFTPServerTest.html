36   * Tests that require a running SSHD server.
37   */
38  public class SFTPServerTest {
39  
40      private SshServer sshd;
41      private int port;
42      private String sftpUserame = "username";
43      private String sftpPassword = "password";
44  
45      /**
46       * https://mina.apache.org/sshd-project/embedding_ssh.html
47       */
48      @Before
49      public void setupSftpServer() throws Exception {
50          try (ServerSocket serverSocket = new ServerSocket(0)) {
51              sshd = SshServer.setUpDefaultServer();
52              this.port = serverSocket.getLocalPort();
53              serverSocket.close();
54              sshd.setPort(this.port);
55              sshd.setKeyPairProvider(new SimpleGeneratorHostKeyProvider(new File("target", "hostkey.ser")));
56              sshd.setPasswordAuthenticator(new PasswordAuthenticator() {
57                  @Override
58                  public boolean authenticate(String username, String password, ServerSession session) throws PasswordChangeRequiredException {
59                      return sftpUserame.equals(username) && sftpPassword.equals(password);
60                  }
61              });
62  
63              sshd.setShellFactory(new ProcessShellFactory("/bin/sh", "-i", "-l"));
64              sshd.setSubsystemFactories(Collections.<NamedFactory<Command>>singletonList(new SftpSubsystemFactory()));
65              sshd.setFileSystemFactory(new VirtualFileSystemFactory(Paths.get(System.getProperty("user.dir"), "target")));
66  
67              sshd.setCommandFactory(new ScpCommandFactory());
68              sshd.start();
69          }
70      }
71  
72      @After
73      public void stopServer() throws Exception {
74          sshd.stop();
75      }
