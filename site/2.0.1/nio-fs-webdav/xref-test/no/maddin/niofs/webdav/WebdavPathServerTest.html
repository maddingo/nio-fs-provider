37      /**
38       * Start an embedded Webdav Server for testing.
39       */
40      @Before
41      public void startWebdavServer() throws Exception {
42          rootFolder = Files.createTempDirectory(Paths.get(".", "target"), "webdav-test").toFile();
43          server = new MiltonWebDAVFileServer(rootFolder);
44          try (ServerSocket ssock = new ServerSocket(0, 1, new InetSocketAddress(0).getAddress())) {
45              webdavPort = ssock.getLocalPort();
46              server.setPort(webdavPort); // optional, defaults to 8080
47              server.getUserCredentials().put(user, password); // optional, defaults to no authentication
48              ssock.close(); // close before the server can re-bind to the random port.
49              server.start();
50          }
51      }
52  
53      /**
54       * Stop the embedded Webdav server.
55       */
56      @After
57      public void stopWebDavServer() throws Exception {
58          server.stop();
59      }

61      /**
62       * Tests {@link Files#createDirectories(Path, FileAttribute[])} with WebDavPath
63       */
64      @Test
65      public void getCreateChildPath() throws Exception {
66          URI uri = new URI("webdav", user + ':' + password,"localhost", webdavPort, "/a/b", null, null);
67          Path path = Paths.get(uri);
68          Path newPath = Files.createDirectories(path);
69          assertThat(newPath, is(notNullValue()));
70          File fileA = new File(rootFolder.getAbsolutePath(), "a");
71          assertThat(fileA.exists(), is(true));
72          assertThat(fileA.isDirectory(), is(true));
73          File fileB = new File(fileA.getAbsolutePath(), "b");
74          assertThat(fileB.exists(), is(true));
75          assertThat(fileB.isDirectory(), is(true));
76      }

78      /**
79       * Tests {@link Files#copy(Path, Path, CopyOption...)} with WebdavPath as target.
80       */
81      @Test
82      public void copyFiles() throws Exception {
83          File src = File.createTempFile("webdavtest", ".txt", new File("target"));
84          try (FileWriter fw = new FileWriter(src)) {
85              fw.append("test test");
86          }
87  
88          URI uriTo = new URI("webdav", user + ':' + password,"localhost", webdavPort, "/webdav/test2/file.txt", null, null);
89  
90          Path pathTo = Paths.get(uriTo);
91          Files.copy(src.toPath(), pathTo, StandardCopyOption.REPLACE_EXISTING);
92  
93          File fileWebdav = new File(rootFolder.getAbsolutePath(), "webdav");
94          assertThat(fileWebdav.exists(), is(true));
95          assertThat(fileWebdav.isDirectory(), is(true));
96          File fileTest2 = new File(fileWebdav.getAbsolutePath(), "test2");
97          assertThat(fileTest2.exists(), is(true));
98          assertThat(fileTest2.isDirectory(), is(true));
99          File fileTxt = new File(fileTest2.getAbsolutePath(), "file.txt");
100         assertThat(fileTxt.exists(), is(true));
101         assertThat(fileTxt.isFile(), is(true));
102 
103         StringBuilder content = new StringBuilder();
104         try (BufferedReader br = new BufferedReader(new FileReader(fileTxt))) {
105             content.append(br.readLine());
106         }
107         assertThat(content.toString(), is(equalTo("test test")));
108     }

121     public void deleteFile() throws Exception {

124         Path path = Paths.get(uri);
125         assumeThat("File should exist prior to deleting it", Files.exists(path), is(true));

142     @Test
143     public void readFileAttributes() throws Exception {
144         File testFile = writeTestFile("testfile.txt");
145 
146         URI uri = new URI("webdav", user + ':' + password, "localhost", webdavPort, "/testfile.txt", null, null);
147 
148         BasicFileAttributes attrs = Files.readAttributes(Paths.get(uri), BasicFileAttributes.class);
149 
150         assertThat(attrs, is(notNullValue()));
151         assertThat(attrs.size(), is(greaterThan(0L)));
152         assertThat(attrs.isDirectory(), is(false));
153         assertThat(attrs.isRegularFile(), is(true));
154         assertThat(attrs.isSymbolicLink(), is(false));
155         assertThat(attrs.isOther(), is(false));
156         assertThat(attrs.lastAccessTime(), is(notNullValue()));
157         assertThat(attrs.lastModifiedTime(), is(notNullValue()));
158         assertThat(attrs.creationTime(), is(notNullValue()));
159     }
