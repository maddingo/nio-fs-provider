1   package no.maddin.niofs.smb;
2   
3   import java.io.IOException;
4   import java.nio.ByteBuffer;
5   import java.nio.channels.SeekableByteChannel;
6   
7   import jcifs.smb.SmbFileOutputStream;
8   
9   /**
10   * Implementation of {@link SeekableByteChannel} for writing SMB files.
11   * 
12   * Reading is not supported at the moment.
13   */
14  public class SMBByteChannel implements SeekableByteChannel {
15  
16    private final SMBPath path;
17    private SmbFileOutputStream out;
18  
19    public SMBByteChannel(SMBPath smbPath) throws IOException {
20      this.path = smbPath;
21      out = new SmbFileOutputStream(path.getSmbFile());
22    }
23  
24    @Override
25    public boolean isOpen() {
26      return out.isOpen();
27    }
28  
29    @Override
30    public void close() throws IOException {
31      out.close();
32    }
33  
34    @Override
35    public int read(ByteBuffer dst) throws IOException {
36      throw new UnsupportedOperationException();
37    }
38  
39    @Override
40    public int write(ByteBuffer src) throws IOException {
41      int len = src.remaining();
42      byte[] buf = new byte[len];
43      while (src.hasRemaining()) {
44        src.get(buf);
45        out.write(buf);
46      }
47      return len;
48    }
49  
50    @Override
51    public long position() throws IOException {
52      throw new UnsupportedOperationException();
53    }
54  
55    @Override
56    public SeekableByteChannel position(long newPosition) throws IOException {
57      throw new UnsupportedOperationException();
58    }
59  
60    @Override
61    public long size() throws IOException {
62      throw new UnsupportedOperationException();
63    }
64  
65    @Override
66    public SeekableByteChannel truncate(long size) throws IOException {
67      throw new UnsupportedOperationException();
68    }
69  }
