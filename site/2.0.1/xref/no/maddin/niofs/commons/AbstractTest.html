11  import org.junit.Assume;

14  /**
15   * Base class for unit tests.
16   * @deprecated try to get rid of this. Test should be self-contained without external configuration.
17   */
18  @Deprecated
19  public abstract class AbstractTest {
20  
21      /**
22       * Test parameters provided by <code>${user.dir}/src/test/nio-test.xml</code>.
23       */
24      protected static Properties testprops = new Properties();
25  
26      @BeforeClass
27      public static void initProps() throws IOException {
28          File testpropsFile = Paths.get(System.getProperty("user.dir"), "src", "test", "nio-test.xml").toFile();
29  
30          Assume.assumeTrue("Can read " + testpropsFile.getAbsolutePath(), testpropsFile.canRead());
31  
32  
33          testprops.loadFromXML(new FileInputStream(testpropsFile));
34      }
35  
36      protected URI createTestUri(String scheme, String host, int port, String path) throws URISyntaxException {
37          String username = getProperty(scheme, "nio.user");
38          String password = getProperty(scheme, "nio.password");
39  
40          return new URI(scheme, username + ':' + password, host, port, path, null, null);
41      }
42  
43      public String getProperty(String schema, String key) {
44          String value = testprops.getProperty(schema+'.'+key);
45          if (value == null) {
46              value = testprops.getProperty(key);
47          }
48          return value;
49      }
50  }
