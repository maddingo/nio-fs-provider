105     @ParameterizedTest
106     @MethodSource("validPathData")
107     public void validPath(URI uri, Matcher<Path> expectedResult) throws IOException {
108         try (FileSystem fs = FileSystems.newFileSystem(serverUri(uri), Collections.emptyMap())) {
109             assertThat(fs, hasProperty("open", equalTo(true)));
110             Path path = Paths.get(uri);
111             assertThat(path, is(expectedResult));
112         }
113     }
114 
115     @ParameterizedTest
116     @MethodSource("normalizeData")
117     void normalize(URI uri, Matcher<Path> normalizedResult, Matcher<Path> parentMatcher) throws IOException {
118         try (FileSystem fs = FileSystems.newFileSystem(serverUri(uri), Collections.emptyMap())) {
119             assertThat(fs, hasProperty("open", equalTo(true)));
120             Path path = Paths.get(uri);
121             Path normalizedPath = path.normalize();
122             assertThat(normalizedPath, is(normalizedResult));
123             Path parentPath = path.getParent();
124             assertThat(parentPath, is(parentMatcher));
125         }
126     }

155     @NotNull
156     private static TypeSafeDiagnosingMatcher<Path> pathStartsWith(String startsWith) {
157         return new TypeSafeDiagnosingMatcher<Path>() {
158             @Override
159             protected boolean matchesSafely(Path item, Description mismatchDescription) {
160                 return Condition.matched(item.startsWith(startsWith), mismatchDescription).matching(equalTo(true));
161             }
162 
163             @Override
164             public void describeTo(Description description) {
165                 description.appendText("path that starts with ").appendValue(startsWith);
166             }
167         };
168     }
169 
170     @NotNull
171     private static URI serverUri(URI uri) {
172         try {
173             return new URI(uri.getScheme(), uri.getRawUserInfo(), uri.getHost(), uri.getPort(), null, null, null);
174         } catch (URISyntaxException e) {
175             throw new IllegalArgumentException(String.valueOf(uri));
176         }
177     }
