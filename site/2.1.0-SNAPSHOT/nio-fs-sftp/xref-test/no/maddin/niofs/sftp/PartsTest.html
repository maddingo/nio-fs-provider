20   * Tests splitting Path parts.
21   */
22  class PartsTest {
23  
24      static Stream<Arguments> validPathsData() {
25  
26          return Stream.of(
27              Arguments.of("/", Collections.emptyList()),
28              Arguments.of("/aa", Collections.singletonList("aa")),
29              Arguments.of("/aa/bb", Arrays.asList("aa", "bb")),
30              Arguments.of("/aa/../bb", Collections.singletonList("bb")),
31              Arguments.of("/aa/../bb", Collections.singletonList("bb")),
32              Arguments.of("/aa/../bb/cc/../d.txt", Arrays.asList("bb", "d.txt")),
33              Arguments.of(".", Collections.emptyList()),
34              Arguments.of("./", Collections.emptyList()),
35              Arguments.of("", Collections.emptyList()),
36              Arguments.of("aa", Collections.singletonList("aa")),
37              Arguments.of(null, Collections.emptyList())
38          );
39      }
40  
41      static Stream<Arguments> invalidPathsData() {
42          return Stream.of(
43  //            Arguments.of(".", instanceOf(IllegalArgumentException.class)),
44  //            Arguments.of("", instanceOf(IllegalArgumentException.class)),
45  //            Arguments.of(null, instanceOf(IllegalArgumentException.class)),
46  //            Arguments.of("./", instanceOf(IllegalArgumentException.class)),
47  //            Arguments.of("aa", instanceOf(IllegalArgumentException.class)),
48              Arguments.of("/aa/../../bb.txt", instanceOf(IllegalArgumentException.class)),
49              Arguments.of("../", instanceOf(IllegalArgumentException.class))
50          );
51      }
52  
53      @ParameterizedTest
54      @MethodSource({"validPathsData"})
55      void validPaths(String input, List<String> result) {
56          SFTPPath path = new SFTPPath(SFTPHost.SFTP_TEST_HOST, input);
57  
58          assertThat(path, hasProperty("parts", is(equalTo(result))));
59      }
60  
61      @ParameterizedTest
62      @MethodSource({"invalidPathsData"})
63      void invalidPaths(String input, Matcher<Exception> expectedException) {
64          try {
65              SFTPPath path = new SFTPPath(SFTPHost.SFTP_TEST_HOST, input);
66  
67              fail("Call with '" + input + "' should have failed.");
68          } catch (Exception ex) {
69              assertThat(ex, is(expectedException));
70          }
71      }
72  
73      @Test
74      void toStringTest() {
75          SFTPHost host = new SFTPHost(new SFTPFileSystemProvider(), URI.create("sftp://localhost"));
76          SFTPPath path = new SFTPPath(host, "/aa/bb/cc.txt");
77  
78          assertThat(path.toString(), is(equalTo("sftp://localhost/aa/bb/cc.txt")));
79      }
