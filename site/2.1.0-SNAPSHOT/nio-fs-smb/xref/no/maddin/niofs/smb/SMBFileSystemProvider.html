58      @Override
59      public FileSystem newFileSystem(URI uri, Map<String, ?> env) throws IOException {
60          SMBParts parts = splitURI(uri);
61          if (!parts.path.isEmpty()) {
62              throw new IllegalArgumentException("The uri should be smb://" + parts.host + SMBShare.SMBFS_SEPARATOR + parts.share + " optional user info");
63          }
64          URI shareUri = shareUri(parts);
65          SMBShare smbShare = new SMBShare(this, parts.host, parts.share, principal(parts.userinfo, env));
66          fileSystems.put(shareUri, smbShare);
67          return smbShare;
68      }

70      /**
71       * package private for testing purposes
72       */
73      static SMBShare.UsernamePassword principal(String userInfo, Map<String, ?> env) {
74  
75          return Optional.ofNullable(userInfo)
76              .flatMap(upwd -> {
77                  Matcher matcher = USERINFO_PATTERN.matcher(upwd);
78                  if (matcher.matches()) {
79                      String domain = matcher.group("domain");
80                      String user = matcher.group("user");
81                      String password = matcher.group("pwd");
82                      return Optional.of(new SMBShare.UsernamePassword(user, password, domain));
83                  } else {
84                      return Optional.empty();
85                  }
86              })
87              .orElseGet(() ->
88                  Optional.ofNullable(env)
89                  .map(m -> {
90                      String username = (String)m.get("USERNAME");
91                      String password = (String) m.get("PASSWORD");
92                      String domain = (String) m.get("DOMAIN");
93                      return new SMBShare.UsernamePassword(username, password, domain);
94                  }).orElse(null)
95              );
96      }

98      SMBParts splitURI(URI uri) {
99          String path = uri.getPath();
100         if (path == null || path.length() < 2 || path.indexOf('/') != 0) {
101             throw new IllegalArgumentException("incorrect share in path " + path);
102         }
103         int shareEndPos = path.indexOf('/', 1);
104         if  (shareEndPos == -1) {
105             shareEndPos = path.length();
106         }
107         String share = null;
108         try {
109             share = URLDecoder.decode(path.substring(1, shareEndPos), StandardCharsets.UTF_8.name());
110             String file = URLDecoder.decode(path.substring(shareEndPos), StandardCharsets.UTF_8.name());
111             return new SMBParts(uri.getHost(), uri.getPort(), uri.getUserInfo(), share, file);
112         } catch (UnsupportedEncodingException e) {
113             throw new RuntimeException(e);
114         }
115     }

117     @Override
118     public FileSystem getFileSystem(URI uri) {
119         SMBParts parts = splitURI(uri);
120 
121         SMBShare share;
122         try {
123             share = fileSystems.get(shareUri(parts));
124             if (share == null) {
125                 throw new FileSystemNotFoundException();
126             }
127             return share;
128         } catch (IOException e) {
129             throw (FileSystemNotFoundException)new FileSystemNotFoundException().initCause(e);
130         }
131     }
132 
133     private URI shareUri(SMBParts parts) throws IOException {
134         try {
135             return new URI(this.getScheme(), parts.userinfo, parts.host, parts.port, '/' + parts.share, null, null);
136         } catch (URISyntaxException e) {
137             throw new IOException(e);
138         }
139     }

141     @Override
142     public Path getPath(URI uri) {
143         SMBShare share = (SMBShare) getFileSystem(uri);
144 
145         SMBParts parts = splitURI(uri);
146         return new SMBPath(share, parts.path);
147     }

171     @Override
172     public void delete(Path path) {
173         FileAllInformation fileInformation = ((SMBShare) smbPath(path).getFileSystem()).getDiskShare().getFileInformation(path.toString());
174         if (fileInformation.getStandardInformation().isDirectory()) {
175             ((SMBShare) smbPath(path).getFileSystem()).getDiskShare().rmdir(path.toString(), true);
176         } else {
177             ((SMBShare) smbPath(path).getFileSystem()).getDiskShare().rm(path.toString());
178         }
179     }
