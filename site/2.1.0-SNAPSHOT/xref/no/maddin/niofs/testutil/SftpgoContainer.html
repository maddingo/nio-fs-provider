14  public class SftpgoContainer extends GenericContainer<SftpgoContainer> implements BasicTestContainer {
15      private static final int WEBDAV_PORT = 8088;
16      private static final int SFTP_PORT = 2022;
17      public static final String USERNAME = "user";
18  
19      @SuppressWarnings("java:S2068")
20      private static final String PASSWORD = "secret";
21  
22      public SftpgoContainer(String testDataResource) {
23          super("drakkan/sftpgo:v2.5.5");
24          this
25              .withCreateContainerCmdModifier(cmd -> cmd.withUser(userString()))
26              .withCommand(
27                  "sftpgo",
28                  "portable",
29                  "--directory",
30                  "/srv/sftpgo",
31                  "--username",
32                  USERNAME,
33                  "--password",
34                  PASSWORD,
35                  "--sftpd-port",
36                  String.valueOf(SFTP_PORT),
37                  "--webdav-port",
38                  String.valueOf(WEBDAV_PORT),
39                  "--permissions",
40                  "*",
41                  "--ssh-commands",
42                  "*",
43                  "--log-level",
44                  "debug"
45              )
46              .withClasspathResourceMapping(testDataResource, "/srv/sftpgo", BindMode.READ_WRITE)
47              .withClasspathResourceMapping("/sftpgo-config", "/var/lib/sftpgo", BindMode.READ_WRITE)
48              .withExposedPorts(WEBDAV_PORT, SFTP_PORT)
49              .waitingFor(Wait.forHttp("/").forPort(WEBDAV_PORT).withBasicCredentials(USERNAME, PASSWORD).forStatusCode(207))
50              .waitingFor(Wait.forListeningPorts(SFTP_PORT))
51          ;
52      }
53  
54      @SuppressWarnings("java:S112")
55      @NotNull
56      private static String userString() {
57          if (!System.getProperty("os.name").toLowerCase().contains("linux")) {
58              String uid = System.getenv("UID");
59              String gid = System.getenv("GID");
60              if (uid != null && gid != null) {
61                  return uid + ":" + gid;
62              }
63              throw new IllegalArgumentException("Set environment variables UID and GID to the current user");
64          }
65          Path procSelf = Paths.get("/proc/self");
66          try {
67              Object uid = Files.getAttribute(procSelf, "unix:uid");
68              Object gid = Files.getAttribute(procSelf, "unix:gid");
69              return uid + ":" + gid;
70          } catch (IOException e) {
71              throw new RuntimeException(e);
72          }
73      }
74  
75      public int getWebdavPort() {
76          return getMappedPort(WEBDAV_PORT);
77      }
78  
79      public String getWebdavUrl() {
80          return "webdav://" + USERNAME + ':' + PASSWORD + '@' + getHost() + ":" + getWebdavPort();
81      }
82  
83      @Override
84      public URI getBaseUri(String protocol) {
85          switch (protocol) {
86              case "webdav":
87                  return URI.create(getWebdavUrl());
88              case "sftp":
89                  return URI.create("sftp://" + USERNAME + ':' + PASSWORD + '@' + getHost() + ":" + getMappedPort(SFTP_PORT));
90              default:
91                  throw new IllegalArgumentException("Unsupported protocol " + protocol);
92          }
93      }
94  }
