68      /**
69       * Get a SFTP Host with the given host, user, password and port.
70       *
71       * @param uri    valid URI
72       * @param create if {@code true} a new SFTPHost is created if none is registered.
73       */
74      private SFTPHost getSFTPHost(URI uri, boolean requireEmptyPath, boolean create) throws URISyntaxException {
75          URI serverUri = SFTPHost.getServerUri(uri, requireEmptyPath);
76  
77          SFTPHost fs = hosts.computeIfAbsent(serverUri, u -> {
78              if (create) {
79                  return new SFTPHost(this, u);
80              } else {
81                  return null;
82              }
83          });
84          if (fs == null) {
85              throw new FileSystemNotFoundException(uri.toString());
86          }
87          return fs;
88      }

95      @Override
96      public DirectoryStream<Path> newDirectoryStream(Path dir, Filter<? super Path> filter) throws IOException {
97          if (!(dir instanceof SFTPPath)) {
98              throw new IllegalArgumentException(dir.toString());
99          }
100 
101         SFTPHost sftpHost = (SFTPHost) dir.getFileSystem();
102 
103         try (SFTPSession sftpSession = new SFTPSession(sftpHost, jsch)) {
104             Vector<ChannelSftp.LsEntry> ls = sftpSession.sftp.ls(((SFTPPath)dir).getPathString());
105 
106             List<Path> list = ls.stream()
107                 .map(ChannelSftp.LsEntry::getFilename)
108                 .filter(fn -> !fn.equals(".") && !fn.equals("..")) // TODO relative filenames not supported
109                 .map(fn -> "/" + fn) // TODO relative filenames not supported
110                 .map(fn -> new SFTPPath(sftpHost,fn))
111                 .filter(p -> {
112                     try {
113                         if (filter != null) {
114                             return filter.accept(p);
115                         } else {
116                             return true;
117                         }
118                     } catch (IOException e) {
119                         throw new RuntimeException(e);
120                     }
121                 })
122                 .collect(Collectors.toList());
123 
124             return new SftpDirStream(list);
125 
126         } catch (JSchException | SftpException e) {
127             throw new IOException(e);
128         }
129     }

131     @Override
132     public void createDirectory(Path dir, FileAttribute<?>... attrs) throws IOException {
133         if (!(dir instanceof SFTPPath)) {
134             throw new IllegalArgumentException(dir.toString());
135         }
136 
137         SFTPHost sftpHost = (SFTPHost)dir.getFileSystem();
138 
139         try (SFTPSession sftpSession = new SFTPSession(sftpHost, jsch)) {
140             List<String> parts = ((SFTPPath) dir).getParts();
141             // remove the first part if it is the root directory (empty string)
142             if (!parts.isEmpty() && "".equals(parts.get(0))) {
143                 parts = parts.subList(1, parts.size()-1);
144             }
145             // Implementation might not support recursive creation of directories
146             for (String subPath : parts) {
147                 try {
148                     sftpSession.sftp.mkdir(subPath);
149                 } catch(SftpException e) {
150                     throw new IOException(subPath, e);
151                 }
152 
153             }
154 
155         } catch (JSchException e) {
156             throw new IOException(e);
157         }
158     }

165     @Override
166     public void copy(Path source, Path target, CopyOption... options) throws IOException {
167         if (!(source instanceof SFTPPath && target instanceof SFTPPath)) {
168             throw new UnsupportedOperationException("Both source and target must be associated with the same provider");
169         }
170       if (source.getFileSystem().equals(target.getFileSystem())) {
171             String sourcePath = ((SFTPPath)source).getPathString();
172             String targetPath = ((SFTPPath)target).getPathString();
173             SFTPHost host = (SFTPHost) source.getFileSystem();
174             copySameProvider(host, sourcePath, targetPath, options);
175         } else {
176             throw new UnsupportedOperationException("Copy between different filesystems not supported");
177         }
178     }
179 
180     private void copySameProvider(SFTPHost host, String source, String target, CopyOption[] options) throws IOException {
181         try (SFTPSession sftpSession = new SFTPSession(host, jsch)) {
182             if (options != null && options.length > 0) {
183                 log.info("Copy option is ignored");
184             }
185             File tmpFile = createTempFile();
186             try (
187                 OutputStream tmpOut = Files.newOutputStream(tmpFile.toPath())
188             ) {
189                 sftpSession.sftp.get(source, tmpOut);
190             }
191             sftpSession.sftp.put(tmpFile.getAbsolutePath(), target);
192         } catch (JSchException | SftpException e) {
193             throw new IOException(e);
194         }
195 
196     }

198     private static File createTempFile() throws IOException {
199         boolean isPosixFS = FileSystems.getDefault().supportedFileAttributeViews().contains("posix");
200         File tempFile = isPosixFS ? createPosixTempfile() : creeateNonPosixTempfile();
201         tempFile.deleteOnExit();
202         return tempFile;
203     }
204 
205     private static File createPosixTempfile() throws IOException {
206         File tempFile;
207         FileAttribute<Set<PosixFilePermission>> attr = PosixFilePermissions.asFileAttribute(PosixFilePermissions.fromString("rwx------"));
208         tempFile = Files.createTempFile("prefix", "suffix", attr).toFile();
209         return tempFile;
210     }

212     @SuppressWarnings({"java:S5443", "java:S899"}) // use of createTempFile is ok here
213     private static File creeateNonPosixTempfile() throws IOException {
214         File tempFile;
215         tempFile = Files.createTempFile("prefix", "suffix").toFile();
216         tempFile.setReadable(true, true);
217         tempFile.setWritable(true, true);
218         tempFile.setExecutable(true, true);
219         return tempFile;
220     }

271     private static class SFTPSession implements AutoCloseable {
272         private final Session session;
273         final ChannelSftp sftp;
274 
275         public SFTPSession(SFTPHost host, JSch jsch) throws JSchException {
276             this.session = jsch.getSession(host.getUsername(), host.getHost(), host.getPort());
277             UserInfo userinfo = new SFTPUserInfo(host.getPassword());
278             session.setUserInfo(userinfo);
279             session.connect();
280 
281             this.sftp = (ChannelSftp)session.openChannel(SFTP);
282 
283             sftp.connect();
284         }
285 
286         @Override
287         public void close() {
288             sftp.quit();
289             session.disconnect();
290         }
291     }
