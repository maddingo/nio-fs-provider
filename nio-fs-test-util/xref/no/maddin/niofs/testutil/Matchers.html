3   import org.hamcrest.Condition;
4   import org.hamcrest.Description;
5   import org.hamcrest.Matcher;
6   import org.hamcrest.TypeSafeDiagnosingMatcher;
7   
8   import static org.hamcrest.Condition.matched;
9   import static org.hamcrest.Condition.notMatched;
10  
11  public class Matchers {
12  
13      private Matchers() {
14      }
15  
16      public static <T> TypeSafeDiagnosingMatcher<T> hasStringValue(Matcher<String> stringMatcher) {
17          return new org.hamcrest.TypeSafeDiagnosingMatcher<T>() {
18  
19              @Override
20              public void describeTo(Description description) {
21                  description.appendText("string value matching (").appendDescriptionOf(stringMatcher).appendText(")");
22              }
23  
24              @Override
25              protected boolean matchesSafely(T item, Description description) {
26                  return isNonNull(item, description)
27                      .matching(stringMatcher);
28              }
29  
30              private Condition<String> isNonNull(T item, Description description) {
31                  return item == null ? notMatched() : matched(item.toString(), description);
32              }
33          };
34      }
35  }
