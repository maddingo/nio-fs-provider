5   import java.io.IOException;

7   import java.nio.file.Path;
8   import java.util.Iterator;
9   import java.util.List;
10  import java.util.concurrent.atomic.AtomicBoolean;
11  import java.util.stream.Collectors;
12  
13  /**
14   * Iterates over SMB entries in an SMB directory. 
15   */
16  public class SMBDirectoryStream implements DirectoryStream<Path> {
17  
18      private final SMBPath smbFile;
19      private final java.nio.file.DirectoryStream.Filter<? super Path> filter;
20      private final SMBFileSystemProvider provider;
21      private final AtomicBoolean iteratorReturned = new AtomicBoolean();
22  
23      SMBDirectoryStream(SMBFileSystemProvider provider, SMBPath smbFile, java.nio.file.DirectoryStream.Filter<? super Path> filter) {
24          this.smbFile = smbFile;
25          this.filter = filter;
26          this.provider = provider;
27      }
28  
29      @Override
30      public void close() throws IOException {
31          smbFile.getShare().close();
32      }
33  
34      @Override
35      public Iterator<Path> iterator() {
36          SMBShare share = smbFile.getShare();
37          List<FileIdBothDirectoryInformation> list = share.getDiskShare().list(smbFile.getSmbPath());
38          return list.stream()
39              .map(finf -> new SMBPath(share, smbFile.getSmbPath() + share.getSeparator() + finf.getFileName()))
40              .map(Path.class::cast)
41              .filter(this::accept)
42              .collect(Collectors.toList())
43              .iterator();
44      }
45  
46      /**
47       * handle null filter and IOException
48       */
49      private boolean accept(Path path) {
50          try {
51              return filter == null || filter.accept(path);
52          } catch (IOException e) {
53              throw new RuntimeException(e);
54          }
55      }
56  }
