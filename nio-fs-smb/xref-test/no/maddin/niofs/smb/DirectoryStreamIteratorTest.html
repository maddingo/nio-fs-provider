50      @NotNull
51      private static DirectoryStream.Filter<Path> fileNameEndsWith(String suffix) {
52          return entry -> Optional.ofNullable(entry.getFileName())
53              .map(Path::toString)
54              .filter(f -> f.endsWith(suffix))
55              .isPresent();
56      }

60      void directoryStreamIterator(URI shareUrl, String parentPath, SortedSet<URI> childrenUrls, DirectoryStream.Filter<Path> filter) throws Exception {
61  
62          Map<String, String> env = new HashMap<>();
63          env.put("USERNAME", "admin");
64          env.put("PASSWORD", "admin");
65          FileSystem fileSystem = FileSystems.newFileSystem(shareUrl, env);
66          assertThat(fileSystem, Matchers.is(notNullValue()));
67          Path remotePath = fileSystem.getPath(parentPath);
68          SortedSet<URI> fileNames = new TreeSet<>(Comparator.comparing(URI::toString));
69          try (DirectoryStream<Path> directoryStream = Files.newDirectoryStream(remotePath, filter)) {
70              for (Path path : directoryStream) {
71                  fileNames.add(path.toUri());
72              }
73          }
74  
75          assertThat(fileNames, equalTo(childrenUrls));
76      }
77  
78      private static URI uri(String url) {
79          return URI.create(url);

82      @SafeVarargs
83      private static <T> SortedSet<T> set(T... elements) {
84          SortedSet<T> set = new TreeSet<>(Comparator.comparing(Object::toString));
85          Collections.addAll(set, elements);
86          return set;
87      }
